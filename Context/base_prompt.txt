        self.log.info("API Request :: Creating credentials")
        self.log.info(role_payload)
        response = Request.post(
            config["API_ENDPOINTS"]["ANALYST_ROLE"], role_payload, self.headers
        )
        return response
    
    def revoke_open_api_credentials(self, role_payload):
        self.log.info("API Request :: Revoking credentials")
        self.log.info(role_payload)
        response = Request.put(
            config["API_ENDPOINTS"]["ANALYST_ROLE"], role_payload, self.headers
        )
        return response

        revoking_credentials_payload = {

            "app_name": app_name,

    def get_role_for_open_api(self, display_name):
        role_details = Request.get(
            OpenAPIendpoint.role_api, headers=self.headers
        )
        assert role_details.status_code == 200
        role_details_json = role_details.json()
        role_id = ""
        for value in role_details_json["results"]:
            if value["display_name"] == display_name:
                role_id = str(value["id"])
                break
        return role_id

class Util(object):
    log = cl.customLogger(logging.INFO)

    def sleep(self, sec, info=""):
        """
        Put the program to wait for the specified amount of time
        """
        if info is not None:
            self.log.info("Wait :: '" + str(sec) + "' seconds for " + info)
        try:
            time.sleep(sec)
        except InterruptedError:
            self.log.info("InterruptedError")

    @classmethod
    def getAlphaNumeric(self, length, type="letters"):
        """
        Get random string of characters
        Parameters:
            length: Length of string, number of characters string should have
            type: Type of characters string should have. Default is letters
            Provide lower/upper/digits for different types
        """
        alpha_num = ""
        if type == "lower":
            case = string.ascii_lowercase
        elif type == "upper":
            case = string.ascii_uppercase
        elif type == "digits":
            case = string.digits
        elif type == "mix":
            case = string.ascii_letters + string.digits
        else:
            case = string.ascii_letters
        return alpha_num.join(secrets.choice(case) for _ in range(length))

    @classmethod
    def getUniqueNameList(self, listSize=5, itemLength=None):
        """
        Get a list of valid email ids
        Parameters:
            listSize: Number of names. Default is 5 names in a list
            itemLength: It should be a list containing number of items equal to the listSize
                        This determines the length of the each item in the list -> [1, 2, 3, 4, 5]
        """
        nameList = []
        for i in range(0, listSize):
            nameList.append(self.getUniqueName(itemLength[i]))
        return nameList

    @classmethod
    def verifyTextContains(self, actualText, expectedText):
        """
        Verify actual text contains expected text string
        Parameters:
            expectedList: Expected Text
            actualList: Actual Text
        """
        self.log.info("Actual Text From Application Web UI --> :: " + actualText)
        self.log.info("Expected Text From Application Web UI --> :: " + expectedText)
        if expectedText.lower() in actualText.lower():
            self.log.info("### VERIFICATION: CONTAINS !!!")
            return True
        else:
            self.log.info("### VERIFICATION: DOES NOT CONTAIN !!!")
            return False

        """
